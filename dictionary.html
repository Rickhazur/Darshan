    <script>

        // Select DOM elements

        const searchInput = document.getElementById('search-input');

        const searchButton = document.getElementById('search-button');

        const resultContainer = document.getElementById('result-container');

        const loadingIndicator = document.getElementById('loading');

        const userExampleContainer = document.getElementById('user-example-container');

        const userExampleTextarea = document.getElementById('user-example');

        const submitExampleButton = document.getElementById('submit-example');

        const feedbackContainer = document.getElementById('feedback-container');


        let currentWord = '';


        // Event listeners

        searchButton.addEventListener('click', searchWord);

        searchInput.addEventListener('keypress', function(e) {

            if (e.key === 'Enter') {

                searchWord();

            }

        });


        submitExampleButton.addEventListener('click', submitUserExample);


        // Search for the word definition

        function searchWord() {

            const word = searchInput.value.trim();

            if (word) {

                loadingIndicator.style.display = 'block';

                resultContainer.innerHTML = '';

                userExampleContainer.style.display = 'none';

                feedbackContainer.style.display = 'none';

                fetchDefinition(word);

            }

        }


        // Fetch the definition from the Streamlit app

        function fetchDefinition(word) {

            currentWord = word;

            const apiUrl = 'https://darshan.streamlit.app/get-definition'; // Update this to your Streamlit app URL


            fetch(apiUrl, {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json',

                },

                body: JSON.stringify({ word }),

            })

            .then(response => {

                if (!response.ok) {

                    throw new Error(`Error: ${response.statusText}`);

                }

                return response.json();

            })

            .then(data => {

                loadingIndicator.style.display = 'none';

                if (data.definition) {

                    displayResult(data, word);

                    userExampleContainer.style.display = 'block';

                } else {

                    displayError("Oops! We couldn't find that word. Try another one!");

                }

            })

            .catch(error => {

                loadingIndicator.style.display = 'none';

                displayError("Something went wrong. Please try again later.");

                console.error('Error:', error);

            });

        }


        // Display the word definition and example

        function displayResult(result, word) {

            let html = `

                <div class="word">${word}</div>

                <div class="definition">${result.definition}</div>

                <div class="example"><strong>Example:</strong> ${result.example || "No example available."}</div>

            `;

            resultContainer.innerHTML = html;

        }


        // Display error messages

        function displayError(message) {

            resultContainer.innerHTML = `<div class="error">${message}</div>`;

        }


        // Submit the user's example sentence

        function submitUserExample() {

            const userExample = userExampleTextarea.value.trim();

            if (userExample) {

                loadingIndicator.style.display = 'block';

                feedbackContainer.style.display = 'none';

                getFeedback(userExample);

                userExampleTextarea.value = ''; // Clear the input field

            }

        }


        // Get feedback on the user's example sentence

        function getFeedback(userExample) {

            const apiUrl = 'https://darshan.streamlit.app/submit-example'; // Update this to your Streamlit app URL


            fetch(apiUrl, {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json',

                },

                body: JSON.stringify({

                    word: currentWord,

                    example: userExample

                })

            })

            .then(response => {

                if (!response.ok) {

                    throw new Error(`Error: ${response.statusText}`);

                }

                return response.json();

            })

            .then(data => {

                loadingIndicator.style.display = 'none';

                if (data.feedback) {

                    displayFeedback(data.feedback);

                } else {

                    displayError("Oops! We couldn't generate feedback. Please try again.");

                }

            })

            .catch(error => {

                loadingIndicator.style.display = 'none';

                displayError("Something went wrong. Please try again later.");

                console.error('Error:', error);

            });

        }


        // Display feedback on the user's example

        function displayFeedback(feedback) {

            feedbackContainer.innerHTML = `<h3>Feedback on Your Example:</h3><p>${feedback}</p>`;

            feedbackContainer.style.display = 'block';

        }
